124void Gameplay::DrawMap(std::vector<sf::String> vecTileMap, int nSize)
125{
126 SPlat.setTexture(AssetManager::GetTexture(vecTexture[nSize]));
127 SDoor.setTexture(AssetManager::GetTexture(vecTexture[2 + nSize]));
128
129 for (int i = 0; i < vecTileMap.size(); i++)
130 for (int j = 0; j < vecTileMap[i].getSize(); j++)
131 {
132 if (vecTileMap[i][j] == 'A') // blocks
133 SPlat.setTextureRect(sf::IntRect(0, 0, getTs(), getTs()));
134 if (vecTileMap[i][j] == 'o') // points
135 SPlat.setTextureRect(sf::IntRect(getTs(), 0, getTs(), getTs()));
136 if ((vecTileMap[i][j] == ' ') || (vecTileMap[i][j] == 'D')) // nothing or door
137 continue;
138
139 SPlat.setPosition(j * getTs() - getOffsetX(), i * getTs() - getOffsetY());
140 WWin.draw(SPlat);
141 }
142
143 for (int i = 0; i < vecTileMap.size(); i++)
144 for (int j = 0; j < vecTileMap[i].getSize(); j++)
145 {
146 if ((vecTileMap[i][j] == 'D')) // door
147 {
148 if (!getDoorOpened())
149 SDoor.setTextureRect(sf::IntRect(0, 0, getTs(), 80 - nSize * 40));
150 else if (getDoorOpened())
151 {
152 SDoor.setTextureRect(sf::IntRect(getTs(), 0, getTs(), 80 - nSize * 40));
153 setEndLevel(true);
154 }
155 }
156 else continue;
157
158 SDoor.setPosition(j * getTs() - getOffsetX(), i * getTs() + 20 - nSize * 10 - getOffsetY());
159 WWin.draw(SDoor);
160 }
161}

231void Gameplay::LevelMenu()
232{
233 std::vector<sf::String> vecNameMenu{ "1 " + vecTitles[settingsValues.getLanguage()],
234 "2 " + vecTitles[settingsValues.getLanguage()], "3 " + vecTitles[settingsValues.getLanguage()],

236 GameMenu myMenu(WWin, 300, 250, 100, 120, vecNameMenu);
237 sf::RectangleShape RSPanel;
238 sf::Text TxtExit;
239 bool bMap = true;
240 int nStart = 1, nEnd = 6;
241
242 ReadValues(settingsValues, VALUES);
243 ReadMap(vecTileMap, 1); // assign a first-level map
244
245 // background
246 RSBackground.setSize(sf::Vector2f(nBgWidth, nBgHeight));
247 RSBackground.setTexture(&AssetManager::GetTexture(vecBackground[0]));

249 // menu
250 myMenu.setColorTextMenu(sf::Color::White, sf::Color::Red, sf::Color::Black);
251 myMenu.AlignMenu(2);

255 RSPanel.setPosition(sf::Vector2f(650, 270));
256 RSPanel.setTexture(&AssetManager::GetTexture(BACKGROUND));
257
258 // exit tip
259 TxtExit.setFont(AssetManager::GetFont(FONTH));
260 InitText(TxtExit, 50, 50, vecTitles[2 + settingsValues.getLanguage()], 50, sf::Color::Magenta, 5, sf::Color::Black);
261
262 setTs(25);
263 setOffsetX(-650);
264 setOffsetY(-270);
265
266 std::tuple<sf::RectangleShape, sf::Text> tDrawElements(RSBackground, TxtExit);
267
268 while (WWin.isOpen())
269 {
270 sf::Event EEvent;
271 while (WWin.pollEvent(EEvent))
272 {
273 if ((EEvent.type == sf::Event::KeyPressed) && (EEvent.key.code == sf::Keyboard::Escape)) return;
274
275 if (EEvent.type == sf::Event::KeyReleased)
276 {
277 if (EEvent.key.code == sf::Keyboard::Up)
278 {
279 myMenu.MoveUp();
280 ReadMap(vecTileMap, myMenu.getSelectedMenuNumber() + 1);
281 }
282
283 if (EEvent.key.code == sf::Keyboard::Down)
284 {
285 if ((myMenu.getSelectedMenuNumber() == nEnd + 1) && (nEnd + 1 < 6))
286 {
287 nStart++;
288 nEnd++;
289 }
290
291 myMenu.MoveDown();
292 ReadMap(vecTileMap, myMenu.getSelectedMenuNumber() + 1);
293 }
294
295 if (EEvent.key.code == sf::Keyboard::Return)
296 {
297 Level();
298 return;
299 }
300 }
301 }
302
303 WWin.clear();
304
305 WinDraw(WWin, tDrawElements);
306
307 if (bMap)
308 {
309 WWin.draw(RSPanel);
310 DrawMap(vecTileMap, 1);
311 }
312
313 myMenu.Draw();
314
315 WWin.display();
316 }

322void Gameplay::Level()
323{
324 bGameplayState = true;
325 Player stick = Player(WWin, vecTileMap, vecSkin[settingsValues.getSkin()], 3 + 2 * settingsValues.getSkin());
326 sf::Clock CClock, CTimer;
327 bool bTimer = false;
328 int nStart = CClock.getElapsedTime().asSeconds(), nFinish;
329
330 // player
331 stick.setTexture(vecSkin[settingsValues.getSkin()]);
332
333 // background
334 RSBackground.setSize(sf::Vector2f(nBgWidth, nBgHeight));
335 RSBackground.setTexture(&AssetManager::GetTexture(vecBackground[3]));
336
337 setTs(50);
338 setOffsetX(0);
339 setOffsetY(0);
340 setPreEx(false);
341 setEndLevel(false);
342
343 while (WWin.isOpen())
344 {
345 sf::Time TDt = CClock.restart();
346
347 setDoorOpened(stick.getDoorOpened());
348
349 Input(stick, bGameplayState);
350 if (!bGameplayState) return;
351 if ((!bPreEx) && (!bEndLevel)) Update(TDt, stick);
352
353 WWin.clear();
354
355 auto drawStick = stick.getStick();
356 std::tuple<sf::RectangleShape, sf::Sprite> tDrawElements(RSBackground, drawStick);
357
358 WinDraw(WWin, tDrawElements);
359
360 Camera(stick, vecTileMap);
361 DrawMap(vecTileMap, 0);
362
363 if (getPreEx()) PreExit();
364 if (getEndLevel()) EndOfTheLevel(nStart, nFinish, CTimer, bTimer);
365
366 WWin.display();
367 }
368}
