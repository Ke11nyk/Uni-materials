50void Animator::Update(sf::Time const& TDt)
51{
52 if (aniCurrentAnimation == nullptr) return;
53
54 TCurrentTime += TDt;
55
56 float scaledTime = (TCurrentTime.asSeconds() / aniCurrentAnimation->TDuration.asSeconds());
57 auto numFrames = static_cast<int>(aniCurrentAnimation->vecFrames.size());
58 auto currentFrame = static_cast<int>(scaledTime * numFrames);
59
60 if (aniCurrentAnimation->bLooping) currentFrame %= numFrames;
61 else
62 if (currentFrame >= numFrames) { currentFrame = numFrames - 1; bEndAnim = true; }
63
64 SSprite.setTextureRect(aniCurrentAnimation->vecFrames[currentFrame]);
65
66}
