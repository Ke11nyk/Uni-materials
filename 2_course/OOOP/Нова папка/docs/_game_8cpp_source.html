161void Game::mainloop()
162{
163 Gameplay myPlay(WWin);
164 sf::Text TxtTitle;
165 std::vector<sf::String> vecNameMenu{ vecTitles[2 + settingsValues.getLanguage()],
166 vecTitles[4 + settingsValues.getLanguage()], vecTitles[6 + settingsValues.getLanguage()],

168 GameMenu myMenu(WWin, 950, 350, 100, 120, vecNameMenu);
169
170 // title

173 200, sf::Color::White, 20, sf::Color::Black);
174
175 // background
176 RSBackground.setSize(sf::Vector2f(nBgWidth, nBgHeight));
177 RSBackground.setTexture(&AssetManager::GetTexture(vecBackground[0]));
178
179 // menu
180 myMenu.setColorTextMenu(sf::Color::White, sf::Color::Red, sf::Color::Black);
181 myMenu.AlignMenu(2);
182
183 std::tuple<sf::RectangleShape, sf::Text> tDrawElements(RSBackground, TxtTitle);
184
185 while (WWin.isOpen())
186 {
187 sf::Event EEvent;
188 while (WWin.pollEvent(EEvent))
189 {
190 if (EEvent.type == sf::Event::KeyReleased)
191 {
192 if (EEvent.key.code == sf::Keyboard::Up) {myMenu.MoveUp(); }
193 if (EEvent.key.code == sf::Keyboard::Down) {myMenu.MoveDown(); }
194 if (EEvent.key.code == sf::Keyboard::Return)
195 {
196 switch (myMenu.getSelectedMenuNumber())
197 {
198 case 0: myPlay.LevelMenu(); break;
199 case 1: Settings(); break;
200 case 2: AboutGame(); break;
201 case 3: WWin.close(); break;
202
203 default: break;
204 }
205 }
206 }
207 }
208
209 WWin.clear();
210
211 WinDraw(WWin, tDrawElements);
212 myMenu.Draw();
213
214 WWin.display();
215 }
