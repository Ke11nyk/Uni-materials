void sort(char num[], char nums[][50], int& koef, int& j)
{	
	int i=0;
	while(num[koef] != ' ')
	    {
	    	if(num[koef] == '\0')
	    	    break;
		    else
		    {
		    	nums[j][i] = num[koef];
		        koef++;
		        i++;
			}	
	    }
	nums[j][i] = '\0';
	
	koef++;
	j++;
}

				bool k = true;
				for(int j=0; j<strlen(nums[n]); j++)

						k = false;
				if(k)
				{
					number += i*koef;
					n++;
				}
			}	
		}
}
